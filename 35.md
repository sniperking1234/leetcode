#35. Search Insert Position
[题目地址](https://leetcode.com/problems/search-insert-position/)
```java
public class Solution {
    public int searchInsert(int[] nums, int target) {
        return dicSearch(nums, 0, nums.length - 1, target);
    }

    /**
     * 二分法查找递归方式
     * @param nums 目标数组
     * @param start 数组开始下标
     * @param end 数组结束下标
     * @param target 目标数字
     * @return 下标，如果找到，则返回所在下标，没找到，则返回应该插入的位置
     */
    public int dicSearch(int[] nums, int start, int end, int target) {
        // start大于end的话，说明没找到，则返回较大的那个作为应当插入的位置
        if (start > end) {
            return start;
        }
        int middle = (start + end) / 2;
        // 看中点是否等于target,如果找到了，则返回下标
        if (nums[middle] == target) {
            return middle;
        }
        // 如果没找到，进行二分查找
        else if (nums[middle] > target) {
            return dicSearch(nums, start, middle - 1, target);
        }
        else {
            return dicSearch(nums, middle + 1, end, target);
        }
    }
}
```

```java
public class Solution {
    public int searchInsert(int[] nums, int target) {
        int start = 0, end = nums.length - 1;
        int middle;
        // 非递归的二分查找
        while (start <= end) {
            middle = (start + end) / 2;
            if (nums[middle] == target)
                return middle;
            else if (nums[middle] < target) {
                start = middle + 1;
            }
            else {
                end = middle - 1;
            }
        }
        return start;
    }
}
```
