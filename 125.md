int isPalindrome(char *s)
{
    int start=0,end = 0;
    while(s[end] != '\0')
    {
        end++;
    }
    while(start != end && start-1 != end)
    {
        if(!((s[start]>='a'&&s[start]<='z')||(s[start]>='A'&&s[start]<='Z')
             ||(s[start]>='0'&&s[start]<='9')))
        {
            start++;
            continue;
        }
        if(!((s[end]>='a'&&s[end]<='z')||(s[end]>='A'&&s[end]<='Z')
             ||(s[end]>='0'&&s[end]<='9')))
        {
            end--;
            continue;
        }
        if(s[start] != s[end] && s[start]+'A'-'a'!=s[end] && s[start]!=s[end]+'A'-'a')
        {
            return 0;
        }
        if((s[start]<='9'&&s[end]>='A')||(s[end]<='9'&&s[start]>='A'))
        {
            return 0;
        }
        start++;end--;
    }
    return 1;
}
